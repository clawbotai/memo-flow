      ↓
  内容理解 → insights.json

  🔧 核心代码结构

  Pipeline 主流程

  // server/pipelines/main.js
  export async function processContent(input, options = {}) {
    const cache = await loadCache(options.taskId);

    // Step 1: 转录（可跳过）
    const transcript = cache.transcript || await transcribe(input.audioUrl);

    // Step 2: 内容理解
    const insights = cache.insights || await analyzeContent(transcript);

    // Step 3: 观点提炼
    const points = cache.points || await extractInsights(insights);

    // Step 4: 爆款加工
    const viralPoints = cache.viralPoints || await viralRewrite(points);

    // Step 5: 内容生成（多平台）
    const outputs = {
      twitter: await generateTwitter(viralPoints),
      redbook: await generateRedbook(viralPoints),
      wechat: await generateWechat(viralPoints)
    };

    // Step 6: 去 AI 味
    const final = await Promise.all(
      Object.entries(outputs).map(async ([platform, content]) => ({
        platform,
        content: await humanize(content)
      }))
    );

    return { transcript, insights, points, viralPoints, outputs: final };
  }

  可重跑机制

  // 支持从任意节点重跑
  export async function rerunStep(taskId, step, input) {
    const cache = await loadCache(taskId);

    const stepMap = {
      'analyze': () => analyzeContent(cache.transcript),
      'extract': () => extractInsights(input || cache.insights),
      'viral': () => viralRewrite(input || cache.points),
      'generate': () => generateContent(input || cache.viralPoints),
      'humanize': () => humanize(input)
    };

    const result = await stepMap[step]();
    await saveCache(taskId, { [step]: result });

    return result;
  }

  数据流设计

  转录完成 (status: completed)
      │
      ▼
  用户切换到"内容创作"Tab
      │
      ▼
  自动/手动触发 Step 1: 内容理解
      │
      ├─ 调用 POST /transcriptions/:id/generate (step: analyze)
      ├─ 后端调用 LLM → 返回 { theme, keyPoints, conclusions }
      ├─ 前端展示 + 支持编辑
      │
      ▼
  Step 2: 观点提炼
      │
      ├─ 传入上一步结果
      ├─ LLM 返回 { viralPoints, controversialPoints, quotes }
      ├─ 前端展示 [☑] 勾选框 + 编辑框
      │
      ▼
  Step 3: 爆款加工
      │
      ├─ 用户勾选要用的观点
      ├─ LLM 返回 { hook, conflict, emotion }
      │
      ▼
  Step 4: 平台生成
      │
      ├─ 用户选择平台 (Twitter/小红书/公众号)
      ├─ LLM 返回对应格式内容
      │
      ▼
  Step 5: 去 AI 味
      │
      └─ 最终输出 → 用户复制/发布

  // helper/lib/content-generator/prompts.js

  // ─────────────────────────────────────────────────────────────
  // Step 1: 内容理解层
  // ─────────────────────────────────────────────────────────────
  function buildAnalyzePrompt(transcript, title) {
    return `你是一个内容分析专家。请分析以下播客/视频转录内容，提取结构化信息。

  转录标题：${title}

  转录内容：
  ${transcript}

  请按以下 JSON 格式返回：
  {
    "theme": "用一句话概括核心主题",
    "audience": "目标受众是谁",
    "contentType": "内容类型（经验分享/观点输出/教学/访谈）",
    "keyPoints": ["关键观点 1", "关键观点 2", "关键观点 3"],
    "conclusions": ["结论 1", "结论 2"]
  }`;
  }

  // ─────────────────────────────────────────────────────────────
  // Step 2: 观点提炼层
  // ─────────────────────────────────────────────────────────────
  function buildExtractPrompt(insights) {
    return `你是一个观点提炼专家。请将以下结构化内容转化为可传播的观点。

  内容主题：${insights.theme}
  关键观点：${insights.keyPoints.join('\n')}
  结论：${insights.conclusions.join('\n')}

  请按以下 JSON 格式返回：
  {
    "viralPoints": ["易于传播的观点 1", "易于传播的观点 2"],
    "controversialPoints": ["有争议/反常识的观点 1", "有争议/反常识的观点 2"],
    "quotes": ["金句 1", "金句 2"]
  }`;
  }

  // ─────────────────────────────────────────────────────────────
  // Step 3: 爆款加工层（核心差异 🔥）
  // ─────────────────────────────────────────────────────────────
  function buildViralPrompt(points) {
    return `你是一个爆款内容专家。请将普通观点改写成具有传播力的爆款内容。

  待加工观点：
  ${points.viralPoints.join('\n')}
  ${points.controversialPoints.join('\n')}

  请按以下 JSON 格式返回：
  {
    "hooks": ["吸引眼球的开场 Hook 1", "吸引眼球的开场 Hook 2"],
    "conflicts": ["制造冲突/反差的表达 1", "制造冲突/反差的表达 2"],
    "emotion": "焦虑|希望|愤怒|共鸣",
    "emotionLines": ["激发情绪的金句 1", "激发情绪的金句 2"]
  }`;
  }

  // ─────────────────────────────────────────────────────────────
  // Step 4: 内容生成层
  // ─────────────────────────────────────────────────────────────
  function buildGeneratePrompt(viralPoints, platform) {
    const platformConfig = {
      twitter: {
        instruction: '生成 Twitter Thread，共 5-8 条推文',
        format: '每条推文独立，有 Hook，有信息增量',
        length: '每条 280 字符以内',
      },
      redbook: {
        instruction: '生成小红书笔记',
        format: '标题 + 正文 + emoji + 标签',
        length: '正文 500-800 字',
      },
      wechat: {
        instruction: '生成公众号文章',
        format: '标题 + 引言 + 正文分段 + 结尾',
        length: '1500-2000 字',
      },
    };

    const config = platformConfig[platform];

    return `你是一个${platform}内容创作专家。${config.instruction}

  爆款素材：
  Hooks: ${viralPoints.hooks.join('\n')}
  冲突点：${viralPoints.conflicts.join('\n')}
  情绪：${viralPoints.emotion}
  情绪金句：${viralPoints.emotionLines.join('\n')}

  要求：
  - ${config.format}
  - ${config.length}
  - 口语化表达
  - 避免 AI 痕迹

  请返回 JSON 格式。`;
  }

  // ─────────────────────────────────────────────────────────────
  // Step 5: 去 AI 味层
  // ─────────────────────────────────────────────────────────────
  function buildHumanizePrompt(content, platform) {
    return `你是一个文字编辑专家。请优化以下内容，使其更像人写的。

  原始内容：
  ${content}

  优化方向：
  1. 打破结构对称（不要每段都一样长）
  2. 增加口语感（用"你"而不是"用户"）
  3. 删除 AI 痕迹（避免"总的来说"、"综上所述"）
  4. 增加个人感（适度使用"我"、"我们"）
  5. 调整节奏（长短句交错）

  直接返回优化后的内容，不要解释。`;
  }

  🏗️ 后端调用结构

  // helper/lib/content-generator/pipeline.js

  async function runContentPipeline(recordId, provider) {
    const record = await getRecord(recordId);
    const settings = await loadLanguageModelSettings();
    const config = getProviderConfig(settings, provider);

    // Step 1: 内容理解
    const analyzePrompt = buildAnalyzePrompt(record.transcript, record.title);
    const insights = await callLLM(config, analyzePrompt);

    // Step 2: 观点提炼
    const extractPrompt = buildExtractPrompt(insights);
    const points = await callLLM(config, extractPrompt);

    // Step 3: 爆款加工
    const viralPrompt = buildViralPrompt(points);
    const viralPoints = await callLLM(config, viralPrompt);

    // Step 4: 内容生成（多平台）
    const outputs = {};
    for (const platform of ['twitter', 'redbook', 'wechat']) {
      const generatePrompt = buildGeneratePrompt(viralPoints, platform);
      outputs[platform] = await callLLM(config, generatePrompt);
    }

    // Step 5: 去 AI 味
    const final = {};
    for (const [platform, content] of Object.entries(outputs)) {
      const humanizePrompt = buildHumanizePrompt(content, platform);
      final[platform] = await callLLM(config, humanizePrompt);
    }

    return {
      insights,
      points,
      viralPoints,
      outputs: final,
    };
  }

  🎯 关键设计点

  1. Prompt 模块化

  每个 Prompt 独立函数，便于：
  - 单独调试
  - 单独重跑
  - A/B 测试不同 Prompt 效果

  2. 上下文传递

  transcript → insights → points → viralPoints → outputs
       ↓           ↓          ↓           ↓
    analyze    extract     viral    generate

  3. 领域化扩展（后续）
